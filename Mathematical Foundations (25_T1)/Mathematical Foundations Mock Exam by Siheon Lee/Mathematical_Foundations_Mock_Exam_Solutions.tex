\textbf{Question 1 (25 marks)}

\begin{enumerate}[label=(\alph*)]
    \item Consider the so-called harmonic series
    \[
    \sum_{n=1}^\infty \frac{1}{n}
    \]
    and show that it does not converge. [2 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}
    
    Group terms:
    \[
    \sum_{n=1}^\infty \frac{1}{n}
    = 1 + \frac12
    + \Bigl(\frac13 + \frac14\Bigr)
    + \Bigl(\frac15 + \frac16 + \frac17 + \frac18\Bigr)
    + \Bigl(\frac19 + \cdots + \frac1{16}\Bigr) + \cdots
    \]
    On each block we lower–bound the terms.
    
    \begin{itemize}
      \item For $\frac13+\frac14$, each term is $\ge \frac14$, so
      \[
      \frac13 + \frac14 > \frac14 + \frac14 = \frac12.
      \]
      \item For $\frac15+\frac16+\frac17+\frac18$, each term is $\ge \frac18$, so
      \[
      \frac15+\frac16+\frac17+\frac18 > 4\cdot\frac18 = \frac12.
      \]
      \item From $1/9$ to $1/16$ there are $8$ terms, each $\ge 1/16$, so the sum is $>8\cdot\frac1{16}=\frac12$.
    \end{itemize}
    
    In general, in the block
    \[
    \sum_{k=2^m+1}^{2^{m+1}} \frac{1}{k}
    \]
    there are $2^m$ terms, each at least $\frac{1}{2^{m+1}}$, so the block sum exceeds $\frac{2^m}{2^{m
    +1}} = \frac{1}{2}$.
    
    Hence, after the first two terms,
    \[
    \sum_{n=1}^\infty \frac{1}{n}
    > 1 + \frac12 + \sum_{m=1}^\infty \frac12 = \infty,
    \]
    so the series diverges.
    \end{mdframed}

    \item Consider the sequence \((x_n)\) defined recursively by  
    \[
    x_1 = 3, \quad x_{n+1} = \frac{1}{3}\left(2x_n + \frac{5}{x_n^2}\right).
    \]
    Show that $x_n > 0$ for all $n$, and then use this to prove that $x_n - x_{n+1} \ge 0$. Conclude that $\lim x_n = \sqrt[3]{5}$.
    You may use the Algebraic Limit Theorem without proof. [10 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}
    
    \textbf{Step 1: Positivity $x_n>0$ (2 marks).}
    We use induction.
    
    Base case: $x_1=3>0$. \hfill (0.5)
    
    Induction step: Assume $x_n>0$. Then $x_n^2>0$ and $5/x_n^2>0$, so
    \[
    2x_n + \frac{5}{x_n^2} > 0
    \quad\Rightarrow\quad
    x_{n+1} = \frac13\left(2x_n + \frac{5}{x_n^2}\right) > 0.
    \]
    Thus $x_n>0$ for all $n$. \hfill (1.5)
    
    \medskip
    \textbf{Step 2: Show $x_n-x_{n+1}\ge0$ (4 marks).}
    
    Compute
    \[
    \begin{aligned}
    x_n - x_{n+1}
    &= x_n - \frac13\left(2x_n + \frac{5}{x_n^2}\right) \\
    &= \frac13\left(3x_n - 2x_n - \frac{5}{x_n^2}\right) \\
    &= \frac13\left(x_n - \frac{5}{x_n^2}\right)
     = \frac13\cdot\frac{x_n^3-5}{x_n^2}.
    \end{aligned}
    \]
    Since $x_n^2>0$, we have
    \[
    x_n - x_{n+1} \ge 0
    \quad\Longleftrightarrow\quad
    x_n^3-5 \ge 0
    \quad\Longleftrightarrow\quad
    x_n^3 \ge 5.
    \]
    
    So it suffices to show $x_n^3\ge5$ for all $n$ (equivalently $x_n\ge\sqrt[3]{5}$ for all $n$).
    
    One acceptable route (any logically complete version gets full marks):
    
    \begin{itemize}
      \item Base: $x_1 = 3 > \sqrt[3]{5}$, so $x_1^3\ge 5$.
      \item Induction step: Assume $x_n\ge\sqrt[3]{5}$. Note that the function
      \[
      f(t) = \frac13\left(2t + \frac{5}{t^2}\right)
      \]
      is decreasing on $(0,\infty)$ for $t\ge\sqrt[3]{5}$ and has local minima at $t=\sqrt[3]{5}$ (see below). Thus $x_{n+1}=f(x_n)\ge f(\sqrt[3]{5}) = \sqrt[3]{5}$.
    \end{itemize}

    Proof that $f(t)$ has critical point at $\sqrt[5]{3}$: $f'(t) = \frac{2}{3} - \frac{10}{3}t^{-3}=0 \Rightarrow t = \sqrt[3]{5}$.
    
    Proof that $\sqrt[5]{3}$ is a local minima: $f''(t) = 10t^{-4}, \quad f''(\sqrt[3]{5}) = \frac{10}{5^{\frac{4}{3}}} > 0$.
    
    Therefore $x_n\ge\sqrt[3]{5}$ for all $n$, and hence $x_n^3\ge 5$ so $x_n-x_{n+1}\ge0$ for all $n$. Thus $(x_n)$ is monotone decreasing.  
    
    Marks: 2 for the difference formula, 2 for a correct argument that $x_n^3\ge5$ for all $n$.
    
    \medskip
    \textbf{Step 3: Boundedness and existence of the limit (2 marks).}
    
    From Step 1, $x_n>0$ and from Step 2 we have $x_n\ge\sqrt[3]{5}$ for all $n$, and $(x_n)$ is decreasing. So $(x_n)$ is bounded below and monotone decreasing, hence convergent by the Monotone Convergence Theorem. Let
    \[
    \lim_{n\to\infty} x_n = L.
    \]
    \hfill (2)
    
    \medskip
    \textbf{Step 4: Identify the limit (2 marks).}
    
    Take limits in the recursion and use the algebra of limits:
    \[
    L = \frac13\left(2L + \frac{5}{L^2}\right).
    \]
    Multiply by 3:
    \[
    3L = 2L + \frac{5}{L^2}
    \quad\Rightarrow\quad
    L = \frac{5}{L^2}
    \quad\Rightarrow\quad
    L^3 = 5.
    \]
    Since $x_n>0$ for all $n$, the limit must be the positive root, so
    \[
    \lim_{n\to\infty} x_n = \sqrt[3]{5}.
    \]
    
    \textbf{Marking scheme (10):}
    \begin{itemize}
      \item 2 marks: proof that $x_n>0$ for all $n$ (induction).
      \item 2 marks: correct computation $x_n-x_{n+1} = \frac13\frac{x_n^3-5}{x_n^2}$.
      \item 2 marks: justification that $x_n^3\ge5$ (or $x_n\ge\sqrt[3]{5}$) for all $n$.
      \item 2 marks: monotone decreasing and bounded below $\Rightarrow$ convergence.
      \item 2 marks: limit equation, solve $L^3=5$ and choose positive root.
    \end{itemize}
    \end{mdframed}

    \item The limit superior $(a_n)$, or $\limsup a_n$ is defined by
    \[
    \limsup a_n = \lim y_n
    \]
    where $y_n = \sup \{ a_k : k \ge n \}$. Similarly, The limit inferior $(a_n)$, or $\liminf a_n$ is defined by
    \[
    \liminf a_n = \lim z_n
    \]
    where $z_n = \inf \{ a_k : k \ge n \}$.
    
    Show that $\liminf a_n = \limsup a_n$ if and only if $\lim a_n$ exists. In this case, all three share the same value. [8 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}
    
    \textbf{Direction 1: If $\lim a_n$ exists then $\liminf a_n = \limsup a_n = \lim a_n$ (4 marks).}

    Assume $\lim a_n = L$.
    
    Letting $y_n = \sup \{ a_k : k \ge n \}$ as usual, I claim that $(\forall n \ge N) \ \vert y_n - a \vert \le \frac{\epsilon}{2}$
    
    Suppose for contradiction that my claim is false. That is, suppose
    \[
    (\exists n \ge N) \ \vert y_n - a \vert > \frac{\epsilon}{2} \Longleftrightarrow a + \frac{\epsilon}{2} < y_n < a - \frac{\epsilon}{2}
    \]
    
    This implies that $y_n > a + \frac{\epsilon}{2}$. Note that the other possibility $y_n < a - \frac{\epsilon}{2}$ is not possible since $(\forall n \in \mathbb{N}) y_n \ge a_n$.
    
    But since $y_n$ is the supremum, $\exists k \ge n$ such that $a_k \ge a + \frac{\epsilon}{2}$ (otherwise $a + \frac{\epsilon}{2}$ would be a smaller upper bound).
    
    But this implies that $\vert a_k - a \vert \ge \frac{\epsilon}{2}$ for some $k \ge n \ge N$, which contradicts how $N$ was previously defined.
    
    Thus, the original claim that $(\forall n \ge N) \ \vert y_n - a \vert \le \frac{\epsilon}{2}$ must be true.
    Since
    \[
    (\forall \epsilon > 0)(\exists N \in \mathbb{N})(\forall n \ge N) \ \vert y_n - a \vert \le \frac{\epsilon}{2} < \epsilon
    \]
    we can now conclude that $y_n \rightarrow a$.

    By using very similar arguments, we can also show that $\liminf a_n = a$.
    
    \medskip
    \textbf{Direction 2: If $\liminf a_n = \limsup a_n$ then $\lim a_n$ exists and equals them (4 marks).}
    
    Assume $\limsup a_n$ and $\liminf a_n$ exist and
    \[
    \limsup a_n = \liminf a_n = L.
    \]
    So $y_n\to L$ and $z_n\to L$.
    
    For all $n$ we have
    \[
    z_n = \inf\{a_k : k \ge n\} \le a_n \le \sup \{a_k : k \ge n\} = y_n.
    \]
    
    We conclude using the Sequence Squeeze Theorem that since $(\forall n \in \mathbb{N}) \ z_n \le a_n \le y_n$ and $y_n \rightarrow L$ and $z_n \rightarrow L$, $a_n \rightarrow L$ as well.
    
    \medskip
    \textbf{Marking scheme (8):}
    \begin{itemize}
      \item 4 marks: ``$\lim a_n$ exists $\Rightarrow$ $\limsup=\liminf=\lim a_n$'' with correct $\varepsilon$ argument for $y_n,z_n$.
      \item 4 marks: converse direction using $z_n\le a_n\le y_n$ and convergence of $y_n,z_n$ to deduce $a_n\to L$ using the Sequence Squeeze Theorem.
    \end{itemize}
    \end{mdframed}
    
    \item Let \((a_n)\) be a bounded sequence defined by  
    \[
    a_n = \cos\left(\frac{n\pi}{4}\right) + \frac{(-1)^n}{n}.
    \]
    Prove or disprove that \((a_n)\) converges. [5 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}
    
    We use the identity that the limit of a sequence converges if and only if all the limits of the subsequence converge to the same limit. In symbols:
    \[
    a_n \rightarrow a \Longleftrightarrow (\text{for every subsequence } a_{n_k}) \ a_{n_k} \rightarrow a
    \]
    
    Examine $\cos\left(\frac{n\pi}{4}\right)$. The values of $\cos\left(\frac{n\pi}{4}\right)$ are periodic:
    \[
    \cos\left(\frac{n\pi}{4}\right) = 1,\ \frac{\sqrt2}{2},\ 0,\ -\frac{\sqrt2}{2},\ -1,\ -\frac{\sqrt2}{2},\ 0,\ \frac{\sqrt2}{2},\ 1,\dots
    \]
    In particular, let us form the subsequences:
    \[
    n_k := 8k, \quad a_{8k} = \cos(8k\cdot \tfrac{\pi}{4}) + \frac{(-1)^{8k}}{8k} = \cos(2k\pi) + \frac{1}{8k} = 1 + \frac{1}{8k},
    \]
    and
    \[
    n_k := 8k+4, \quad a_{8k+4} = \cos\big((8k+4)\tfrac{\pi}{4}\big) + \frac{(-1)^{8k+4}}{8k+4} = \cos(\pi+2k\pi) + \frac{1}{8k+4} = \frac{1}{8k+4} -1.
    \]
    Taking the limit of both subsequences, we find:
    \[
    \lim a_{8k} = \lim \frac{1}{8k} + 1 = 1
    \]
    and
    \[
    \lim a_{8k+4} = \lim \frac{1}{8k+4} -1 = -1.
    \]
    Thus $(a_n)$ has subsequences converging to $1$ and to $-1$ and so does not converge. (2 marks)
    
    \textbf{Marking scheme (5):}
    \begin{itemize}
      \item 3 marks: show $\cos(n\pi/4)$ is non-convergent (e.g.\ by giving subsequences with limits $1$ and $-1$).
      \item 2 marks: deduce non-convergence of $(a_n)$ via subsequences.
    \end{itemize}
    \end{mdframed}
    
\end{enumerate}

\textbf{Question 2 (20 marks)}
\begin{enumerate}[label=(\alph*)]

    \item Let
    \[
    U = \{ (x,y,z) \in \mathbb{R}^3 \mid x+y+z = 0 \}, \quad V = \{(x,y,z) \in \mathbb{R}^3 \mid x-y = 0 \}.
    \]
    Find a basis for $U \cap V$. [5 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}
    
    We need all $(x,y,z)\in\mathbb{R}^3$ that satisfy \emph{both}
    \[
    x+y+z=0,
    \qquad
    x-y=0.
    \]
    
    From $x-y=0$ we get $x=y$. Substituting into $x+y+z=0$:
    \[
    x + x + z = 0 \quad\Rightarrow\quad 2x + z = 0 \quad\Rightarrow\quad z = -2x.
    \]
    
    So every vector in $U\cap V$ has the form
    \[
    (x,x,-2x) = x(1,1,-2), \quad x\in\mathbb{R}.
    \]
    
    Thus
    \[
    U\cap V = \{\,x(1,1,-2): x\in\mathbb{R}\,\},
    \]
    and a basis is, for example,
    \[
    \{(1,1,-2)\}.
    \]
    
    \textbf{Marks (5):}
    \begin{itemize}
      \item 2 marks: correctly sets up and solves the system $x-y=0$, $x+y+z=0$ to obtain $(x,x,-2x)$ or equivalent parametric description.
      \item 2 marks: identifies the intersection as $\{x(1,1,-2):x\in\mathbb{R}\}$ (or an equivalent one‑dimensional subspace).
      \item 1 mark: explicitly states a basis, e.g.\ $\{(1,1,-2)\}$.
    \end{itemize}
    \end{mdframed}
    
    \item State the dimension (number of vectors in basis) of $U \cap V$. [2 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}

    From part (a), a basis for $U\cap V$ is $\{(1,1,-2)\}$, which consists of a single nonzero vector. Therefore
    \[
    \dim(U\cap V) = 1.
    \]
    
    \textbf{Marks (2):}
    \begin{itemize}
      \item 2 marks: correctly identifies that there is exactly one basis vector, hence the dimension is $1$.
    \end{itemize}
    \end{mdframed}
    
    \item Find a basis for $U+V$. [8 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}

    We have
    \[
    U = \{(x,y,z)\in\mathbb{R}^3 : x+y+z=0\},
    \qquad
    V = \{(x,y,z)\in\mathbb{R}^3 : x=y\}.
    \]
    
    \textbf{Step 1: Find bases for $U$ and $V$.}
    
    For $U$: Equation $x+y+z=0$ gives $z=-x-y$, so any vector in $U$ can be written as
    \[
    (x,y,z) = (x,y,-x-y) = x(1,0,-1) + y(0,1,-1).
    \]
    Thus one basis for $U$ is
    \[
    B_U = \{(1,0,-1),(0,1,-1)\}.
    \]
    
    For $V$: Equation $x-y=0$ gives $x=y$. So any vector in $V$ has the form
    \[
    (x,y,z) = (t,t,s) = t(1,1,0) + s(0,0,1),
    \]
    and a basis is
    \[
    B_V = \{(1,1,0),(0,0,1)\}.
    \]
    
    \textbf{Step 2: Find a spanning set for $U+V$.}
    
    By definition,
    \[
    U+V = \{u+v : u\in U,\ v\in V\},
    \]
    and hence $U+V$ is spanned by all vectors in $B_U\cup B_V$:
    \[
    S = \{(1,0,-1),(0,1,-1),(1,1,0),(0,0,1)\}.
    \]
    
    \textbf{Step 3: Extract a basis (remove linear dependence).}
    
    We test for linear independence. Note first that
    \[
    (1,1,0) = (1,0,-1) + (0,1,-1) + 2(0,0,1),
    \]
    since
    \[
    (1,0,-1) + (0,1,-1) + 2(0,0,1) = (1,1,-2+2) = (1,1,0).
    \]
    Therefore $(1,1,0)$ is a linear combination of the other three vectors in $S$, so it is not needed in a basis.
    
    Now check that the remaining three vectors
    \[
    (1,0,-1),\quad (0,1,-1),\quad (0,0,1)
    \]
    are linearly independent. Suppose
    \[
    \alpha(1,0,-1) + \beta(0,1,-1) + \gamma(0,0,1) = (0,0,0).
    \]
    Writing out coordinates:
    \[
    (\alpha, \beta, -\alpha-\beta+\gamma) = (0,0,0).
    \]
    Hence $\alpha=0$, $\beta=0$ and then $-\alpha-\beta+\gamma = \gamma = 0$, so $\gamma=0$. Therefore only the trivial combination gives zero, so they are linearly independent.
    
    Thus a basis for $U+V$ is
    \[
    \{(1,0,-1),(0,1,-1),(0,0,1)\}.
    \]
    
    \textbf{Marks (8):}
    \begin{itemize}
      \item 2 marks: correct basis for $U$, e.g.\ $\{(1,0,-1),(0,1,-1)\}$, with justification.
      \item 2 marks: correct basis for $V$, e.g.\ $\{(1,1,0),(0,0,1)\}$, with justification.
      \item 3 mark: form $S = B_U\cup B_V$ and observe redundancy (e.g.\ express $(1,1,0)$ as combination of the others).
      \item 1 mark: state a correct basis for $U+V$, e.g.\ $\{(1,0,-1),(0,1,-1),(0,0,1)\}$, and note linear independence.
    \end{itemize}
    \end{mdframed}
    
    \item Determine whether $U+V = \mathbb{R}^3$. [4 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}

    From part (c), a basis for $U+V$ has three linearly independent vectors:
    \[
    \{(1,0,-1),(0,1,-1),(0,0,1)\}.
    \]
    Hence
    \[
    \dim(U+V) = 3.
    \]
    
    Since $U+V$ is a subspace of $\mathbb{R}^3$ and has dimension $3$, it spans all directions in $\mathbb{R}^3$. Hence,it must equal $\mathbb{R}^3$. Therefore
    \[
    U+V = \mathbb{R}^3.
    \]
    Remark: this means that $\{(1,0,0),(0,1,0),(0,0,1)\}$ is another basis for $U+V$

    Another way to show this is to show that a linear combination of the basis forms a surjective map onto $\mathbb{R}^3$.
    
    \textbf{Marks (4):}
    \begin{itemize}
      \item 2 marks: correctly computes or states that $\dim(U+V)=3$ based on the basis from (c).
      \item 2 marks: correct conclusion that any $3$‑dimensional subspace of $\mathbb{R}^3$ is all of $\mathbb{R}^3$, so $U+V=\mathbb{R}^3$.
    \end{itemize}
    \end{mdframed}
    
\end{enumerate}

\textbf{Question 3 (20 marks)}
\begin{enumerate}[label=(\alph*)]
    \item Let $V$ be the vector space of all real-valued continuous functions on $\mathbb{R}$ of the form
    \[
    f(x) = acos(x) + bsin(x) + c,
    \]
    where $a,b,c \in \mathbb{R}$.

    Define a mapping $T : V \rightarrow \mathbb{R}^3$ by
    \[
    T(f) = \left[ \begin{array}{c}
        f(0) \\ f\left(\frac{\pi}{2}\right) \\ f(\pi)
    \end{array}\right]
    \]
    Prove that $T$ is a linear map. [5 marks]

    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}

    To show $T$ is linear, we must prove:
    \[
    T(f+g) = T(f) + T(g), \qquad
    T(\alpha f) = \alpha\,T(f)
    \]
    for all $f,g\in V$ and all $\alpha\in\mathbb{R}$. This is known as additivity and homogeneity, respectively.
    
    Let $f,g\in V$. Then for each $x$,
    \[
    (f+g)(x) = f(x)+g(x).
    \]
    Hence
    \begin{flalign*}
        T(f+g) &= \begin{bmatrix}
        (f+g)(0) \\[0.2em]
        (f+g)\!\left(\dfrac{\pi}{2}\right) \\[0.2em]
        (f+g)(\pi)
        \end{bmatrix}
        =
        \begin{bmatrix}
        f(0)+g(0) \\[0.2em]
        f\!\left(\dfrac{\pi}{2}\right)+g\!\left(\dfrac{\pi}{2}\right) \\[0.2em]
        f(\pi)+g(\pi)
        \end{bmatrix} \\
        &=
        \begin{bmatrix}
        f(0) \\[0.2em] f\!\left(\dfrac{\pi}{2}\right) \\[0.2em] f(\pi)
        \end{bmatrix}
        +
        \begin{bmatrix}
        g(0) \\[0.2em] g\!\left(\dfrac{\pi}{2}\right) \\[0.2em] g(\pi)
        \end{bmatrix}
        = T(f)+T(g).
    \end{flalign*}
    
    Similarly, for any $\alpha\in\mathbb{R}$ and $f\in V$,
    \[
    (\alpha f)(x) = \alpha f(x),
    \]
    so
    \[
    T(\alpha f) =
    \begin{bmatrix}
    (\alpha f)(0) \\[0.2em]
    (\alpha f)\!\left(\dfrac{\pi}{2}\right) \\[0.2em]
    (\alpha f)(\pi)
    \end{bmatrix}
    =
    \begin{bmatrix}
    \alpha f(0) \\[0.2em] \alpha f\!\left(\dfrac{\pi}{2}\right) \\[0.2em] \alpha f(\pi)
    \end{bmatrix}
    =
    \alpha
    \begin{bmatrix}
    f(0) \\[0.2em] f\!\left(\dfrac{\pi}{2}\right) \\[0.2em] f(\pi)
    \end{bmatrix}
    = \alpha\,T(f).
    \]
    
    Thus $T$ is linear.
    
    \textbf{Marks (5):}
    \begin{itemize}
      \item 2 marks: correct verification that $T(f+g)=T(f)+T(g)$ using componentwise evaluation.
      \item 2 marks: correct verification that $T(\alpha f)=\alpha T(f)$.
      \item 1 mark: clear statement that both properties hold and therefore $T$ is linear.
    \end{itemize}
    \end{mdframed}
    
    \item Prove that $T$ is bijective. [7 marks]

    \[
    p_A(\lambda) = \det(A - \lambda I_n).
    \]
    
    It is a monic polynomial of degree $n$, and its roots are exactly the eigenvalues of $A$. Hence we can factor
    \[
    p_A(t) = \det(A - \lambda I_n) = (\lambda_1 - \lambda)(\lambda_2 - \lambda) \cdots (\lambda_n - \lambda) = \prod_{i=1}^n (\lambda_i - \lambda),
    \]
    or equivalently
    \[
    p_A(t) = \det(\lambda I_n - A) =  \prod_{i=1}^n (\lambda - \lambda_i),
    \]
    depending on the chosen sign convention.

    \[
    p_A(0) = \det(A - 0 \cdot I_n) = \det(A).
    \]

    \[
    p_A(0) = \prod_{i=1}^n (\lambda_i - 0)
           = \prod_{i=1}^n \lambda_i.
    \]
    
    Thus
    \[
    \det(A) = p_A(0) = \prod_{i=1}^n \lambda_i,
    \]
    which is the desired result.

    \item Show that $\bm{H}$ is idempotent and symmetric. [5 marks]
    
    \begin{mdframed}[linecolor=black, linewidth=1pt]
    \textbf{Solution and marking scheme.}

    Recall
    \[
    \bm{H} = \bm{X}(\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top}.
    \]
    
    \textbf{Idempotent:} $\bm{H}^2=\bm{H}$.
    
    \[
    \begin{aligned}
    \bm{H}^2
    &= \left[ \bm{X}(\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top} \right]
       \left[ \bm{X}(\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top} \right] \\
    &= \bm{X}(\bm{X}^{\top}\bm{X})^{-1}
       \underbrace{\bm{X}^{\top}\bm{X}}_{2\times 2}
       (\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top}\\
    &= \bm{X}(\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top}\\
    &= \bm{H}.
    \end{aligned}
    \]
    
    \textbf{Symmetric:} $\bm{H}^{\top}=\bm{H}$.
    
    \[
    \begin{aligned}
    \bm{H}^{\top}
    &= \left[\bm{X}(\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top}\right]^{\top}\\
    &= \left(\bm{X}^{\top}\right)^{\top}
       \left[(\bm{X}^{\top}\bm{X})^{-1}\right]^{\top}
       \bm{X}^{\top}\\
    &= \bm{X}\left[(\bm{X}^{\top}\bm{X})^{\top}\right]^{-1}\bm{X}^{\top}
      \quad\text{(since $(A^{-1})^{\top}=(A^{\top})^{-1}$)}\\
    &= \bm{X}(\bm{X}^{\top}(\bm{X}^\top)^\top)^{-1}\bm{X}^{\top}\\
    &= \bm{X}(\bm{X}^{\top}\bm{X})^{-1}\bm{X}^{\top}\\
    &= \bm{H},
    \end{aligned}
    \]
    using that $\bm{X}^{\top}\bm{X}$ is symmetric.
    
    \textbf{Marks (5):}
    \begin{itemize}
      \item 3 marks: correct computation showing $\bm{H}^2 = \bm{H}$ (idempotence).
      \item 2 marks: correct transpose argument showing $\bm{H}^{\top} = \bm{H}$ (symmetry).
    \end{itemize}
    \end{mdframed}

\textbf{2.} (20 marks: 10 marks explanation, 10 marks code)

Find and classify the critical points of the function
\[
f(x,y) = 2x^3 - 3xy^2 + 6x - y
\]
using either R or Python. Explain your answers using the eigenvalues of the Hessian.

\begin{mdframed}[linecolor=black, linewidth=1pt]
\textbf{Solution and marking scheme (analytical explanation, 10 marks).}

To find the critical points, we first compute the first-order partial derivatives of $f$ with respect to $x$ and $y$, set them equal to zero, and solve for $x$ and $y$.
\[
f_x = \frac{\partial f}{\partial x} = 6x^2 - 3y^2 + 6 = 0, \qquad
f_y = \frac{\partial f}{\partial y} = -6xy - 1 = 0.
\]

From $f_y=0$:
\[
-6xy - 1 = 0 \quad\Rightarrow\quad xy = -\frac{1}{6}.
\]
From $f_x=0$:
\[
6x^2 - 3y^2 + 6 = 0 \quad\Rightarrow\quad 2x^2 - y^2 + 2 = 0
\quad\Rightarrow\quad y^2 = 2x^2 + 2.
\]

Using $xy=-1/6$, we can write $y = -\dfrac{1}{6x}$ (assuming $x\neq 0$, which will be confirmed). Substitute into $y^2=2x^2+2$:
\[
\frac{1}{36x^2} = 2x^2 + 2.
\]

Multiply by $36x^2$:
\[
1 = 72x^4 + 72x^2.
\]
Let $t=x^2\ge 0$, then
\[
72t^2 + 72t - 1 = 0.
\]
Solve:
\[
t = \frac{-72 \pm \sqrt{72^2 - 4\cdot 72\cdot (-1)}}{2\cdot 72}
= \frac{-72 \pm \sqrt{5184 + 288}}{144}
= \frac{-72 \pm \sqrt{5472}}{144}.
\]
Note $\sqrt{5472}=24\sqrt{38}$, so
\[
t = \frac{-72 \pm 24\sqrt{38}}{144}
  = \frac{-3 \pm \sqrt{38}}{18}.
\]
Since $t=x^2\ge0$, we must take the plus sign:
\[
x^2 = \frac{-3 + \sqrt{38}}{18} > 0.
\]
Thus $x = \pm \sqrt{\dfrac{-3 + \sqrt{38}}{18}}$, and from $y=-1/(6x)$ we get corresponding $y$ values. Hence we obtain two critical points:
\[
(x_1,y_1) = \left(\sqrt{\frac{-3 + \sqrt{38}}{18}},\ -\frac{1}{6x_1}\right),
\quad
(x_2,y_2) = \left(-\sqrt{\frac{-3 + \sqrt{38}}{18}},\ -\frac{1}{6x_2}\right).
\]

\medskip
\textbf{Hessian and classification by eigenvalues.}

Compute second derivatives:
\[
f_{xx} = \frac{\partial^2 f}{\partial x^2} = 12x,\qquad
f_{yy} = \frac{\partial^2 f}{\partial y^2} = -6x,\qquad
f_{xy} = f_{yx} = -6y.
\]

The Hessian matrix is
\[
H(x,y) =
\begin{bmatrix}
12x & -6y\\
-6y & -6x
\end{bmatrix}.
\]

At a critical point $(x_c,y_c)$, the eigenvalues of $H(x_c,y_c)$ determine the type:
\begin{itemize}
  \item both eigenvalues $>0$: local minimum;
  \item both eigenvalues $<0$: local maximum;
  \item eigenvalues of opposite sign: saddle point.
\end{itemize}

One may compute eigenvalues analytically (messy) or numerically in R/Python. The trace and determinant are:
\[
\operatorname{tr}H = 12x_c - 6x_c = 6x_c,\qquad
\det H = (12x_c)(-6x_c) - (-6y_c)^2 = -72x_c^2 - 36y_c^2.
\]
Using $y_c^2 = 2x_c^2 + 2$, we get
\[
\det H = -72x_c^2 - 36(2x_c^2+2) = -144x_c^2 - 72 < 0.
\]
So at every critical point the determinant of $H$ is negative, implying that the eigenvalues have opposite signs. Therefore, \emph{both} critical points are \textbf{saddle points}.

\begin{itemize}
  \item 3 marks: correct first derivatives $f_x$, $f_y$ and system $f_x=f_y=0$.
  \item 3 marks: correct reduction to equations $xy=-1/6$, $y^2=2x^2+2$, and solving to obtain two real critical points (an explicit numerical approximation is acceptable).
  \item 2 marks: correct Hessian $H(x,y)$.
  \item 2 marks: uses $\det H<0$ at both critical points to conclude they are saddle points (or equivalently, shows eigenvalues of opposite sign).
\end{itemize}

Any correct script that:

  \item solves $f_x=f_y=0$ for the critical points,
  \item computes the Hessian at each critical point,
  \item computes eigenvalues and classifies each point,
\end{enumerate}

\begin{verbatim}
import sympy as sp

# Symbols
x, y = sp.symbols('x y', real=True)

# Function
f = 2*x**3 - 3*x*y**2 + 6*x - y

# First derivatives
fx = sp.diff(f, x)
fy = sp.diff(f, y)

# Solve for critical points
crit_points = sp.solve([fx, fy], (x, y), dict=True)

print("Critical points:")
for cp in crit_points:
    print(cp)

# Second derivatives and Hessian
f_xx = sp.diff(fx, x)
f_xy = sp.diff(fx, y)
f_yy = sp.diff(fy, y)

H = sp.Matrix([[f_xx, f_xy],
               [f_xy, f_yy]])

for cp in crit_points:
    H_cp = H.subs(cp)
    eigvals = H_cp.eigenvals()
    print("\nAt critical point:", cp)
    print("Hessian:\n", H_cp)
    print("Eigenvalues:", eigvals)

    # Classification from eigenvalues
    eig_list = list(eigvals.keys())
    if eig_list[0] > 0 and eig_list[1] > 0:
        print("Classification: local minimum")
    elif eig_list[0] < 0 and eig_list[1] < 0:
        print("Classification: local maximum")
    else:
        print("Classification: saddle point")
\end{verbatim}
